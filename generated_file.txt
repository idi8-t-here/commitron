                "Mon" => Ok(Date::Mon),

                "Apr" => Ok(Month::Apr),
                "May" => Ok(Month::May),

                "Sep" => Ok(Month::Sep),
                "Oct" => Ok(Month::Oct),
                "Nov" => Ok(Month::Nov),
                "Dec" => Ok(Month::Dec),
                _ => Err(()),
            }
        }
    }
